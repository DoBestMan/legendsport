TOURNAMENTS
    id                      integer
    avatar                  boolean                         Indica si tiene avatar. Se hace referencia a un archivo con el ID de nombre en una ubicación de users/avatars
    name                    string                         Nombre del torneo
    type                    enum                            Si el torneo es de un solo deporte o varios
        Single (Torneo con solo un deporte en sus eventos)
        Multiple (Torneos con más de un deporte en sus eventos)
    ¿sport/sports?
    players_limit           enum                            Cantidad mínima y maxima de usuarios que pueden registrarse. Al cumplir el registro, el estado cambia a FULL
        Heads-Up => 2
        Full => 10
        Unlimited => Unlimited (0)
    buy_in                  smallInteger                    Valor del registro en el torneo
    chips                   mediumInteger                   Fichas que se dan al registrarse en un evento. Al crear se establece config.default_chips
    commission              mediumInteger                   Comisión que cobra la app por registrarse en un evento. Al crear se establece config.default_commission
    late_register           boolean                         TRUE permite registrarse luego de que el torneo inició
    late_register_rule      json                            Regla con el tiempo de registro tardío. Igual al formato PHP para aumentar tiempo
        {
            interval: d/m,
            value: number
        }
    prize_pool              json                            Premio y modo
        {
            type: auto/fixed                                AUTO: El premio será el total del valor de ingreso de cada jugador / FIXED: el valor es fijo
            fixed: 0                                        Valor del premio fijo
        }
    prizes                  json                            Repartición de premiación y modo
        {
            type: auto/fixed                                AUTO: La repartición se hace siguiendo una tabla de porcentajes / FIXED: se fija
        }
    state                   enum                            Estado del torneo
        Announced (En anuncio)
        Registering (En registro)
        Late registering (En registro tardio)
        Running (En ejecución)
        Completed (Terminado)
        Cancel (Cancelado)
    created_at
    updated_at
    user_admin_id
